4、示例练习：数组元素逆置  
```cpp
#include<iostream>
using namespace std;

int main() {
	
	//1、创建一个数组
	int arr[5] = { 1,2,3,4,5};

	int start = 0;
	int end = sizeof(arr) / sizeof(arr[0]) - 1;
	while(start < end) {

		int temp = arr[start];
		arr[start] = arr[end];
		arr[end] = temp;
		start++;
		end--;
	}
	cout << "逆序之后的数组为" << endl;
	for (int i = 0; i < 5; i++)
	{
		cout << arr[i] << endl;
	}
	return 0;
}
```
这个示例主要运用了逆序的这个算法，首先是末尾数组中的数的表达（sizeof(arr) / sizeof(arr[0]) - 1），再有就是逆序的这个过程要写入循环语句中，还有就是这个循环语句的这个条件，什么时候这个内容截止，接着还有一点就是注意一下这个数组是怎么输出的，在其中使用了for的这个循环  
5、冒泡排序  
基本思想：通过相邻元素的比较和交换，将较大的元素逐渐"冒泡"到数列的末端  
核心操作：比较相邻的两个元素，如果前一个比后一个大，就交换它们  
重复过程：对数列进行多次遍历，每次遍历都会将当前未排序部分的最大元素放到正确位置  
```cpp
#include<iostream>
using namespace std;

int main() {
	
	//1、创建一个数组
	int arr[9] = { 4,2,8,0,5,7,1,3,9};
	cout << "排序前的结果" << endl;
	for (int i = 0; i < 9; i++) {
		cout << arr[i] << " ";
	}
    cout << endl;
	//2、冒泡排序
	//开始排序的轮数等于数组元素个数-1
	for (int i = 0; i < 9 - 1; i++) {
		//内层循环，每轮对比的次数=数组元素个数-排序的轮数-1
		for (int j = 0; j < 9 - i - 1; j++) {
			if (arr[j] > arr[j + 1]) {
				int temp = arr[j];
				arr[j] = arr[j + 1];
				arr[j + 1] = temp;
			
			}
		}
	}
	cout << "排序后的结果" << endl;
	for (int i = 0; i < 9; i++) {
		cout << arr[i] << " ";
	}
	return 0;
}
```
总督来说冒泡排序要注意的就是外层循环的次数为元素的个数-1，内层循环的次数为元素的个数-当前循环的轮数-1，这样前他嵌套循环之后再使用换位的三行简单的代码就可以实现冒泡排序了，我觉得最核心的部分就是要找出外层循环与内层循环分别的次数

2、二维数组名  
查看二维数组所占用的内存空间  
获取二维数组的首地址  
```cpp
#include<iostream>
using namespace std;

int main() {

	int arr[2][3] = {
		{1,2,3},
		{4,5,6 }
	};
	cout << sizeof(arr[0]) << endl;
	cout << "二维数组的行数" << sizeof(arr) / sizeof(arr[0]) << endl;
	cout << "二维数组的列数" << sizeof(arr[0]) / sizeof(arr[0][0])<< endl;
	cout<<"二维数组的首地址"<<(int)arr<<endl;
    cout<<"二维数组的第一行元素首地址"<<(int)arr[0]<<endl;
    cout<<"二维数组的第二行元素首地址"<<(int)arr[1]<<endl;
    cout<<"二维数组的第一行第一个元素首地址"<<(int)&arr[0][0]<<endl;//此时需要取址符号&
	return 0;
}
```
3、二维数组案例  
```cpp
#include<iostream>
using namespace std;

int main() {

	int arr[3][3] = {
		{100,100,100},
		{90,50,100 },
		{60.70,80} 
	};
	for (int i = 0; i < 3; i++)
	{
		int sum = 0;
		for (int j = 0; j < 3; j++)
		{
			sum += arr[i][j];
		}
		cout << "第" << i + 1 << "人的分数为" << sum << endl;

	}
	
	return 0;
}
```
